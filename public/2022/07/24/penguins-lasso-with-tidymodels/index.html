3 A bit more than minimal

rec1 <-
  recipe(body_mass_g ~ ., data = penguins) %>% 
  step_dummy(all_nominal()) %>% 
  step_normalize(all_numeric_predictors()) %>% 
  step_nzv(all_numeric_predictors()) %>% 
  step_naomit(all_predictors())
