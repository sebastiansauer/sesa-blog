show_cols <- function(col_pal){
d <- tibble(x = 1:length(col_pal),
            col = factor(col_pal))

ggplot(d) +
  aes(x=x)+
  geom_bar(aes(fill = col)) +
  theme_void() +
  scale_fill_manual(values = col_pal)
}
